// =====================
// SYSTEM STATE
// =====================
struct SystemState {
  float o2Value = 0.0;
  float setpoint = 20.9;
  bool relayOn = false;
  bool sensorSleeping = false;

  bool lockoutActive = false;
  unsigned long lockoutStart = 0;

  bool manualOverride = false;
  bool manualRelayState = false;
};

SystemState state;

// =====================
// TIMING
// =====================
unsigned long lastReadTime = 0;
const unsigned long readInterval = 10000; // 10 seconds

const unsigned long lockoutTime = 1800000UL; // 30 minutes

// Manual override timeout
unsigned long manualStartTime = 0;
const unsigned long manualTimeout = 300000UL; // 5 minutes

// SENSOR BUFFER

char o2_concentration[32];

// =====================
// AUTH CHECK
// =====================
bool checkAuth() {
  if (!server.authenticate(www_user, www_pass)) {
    server.requestAuthentication();
    return false;
  }
  return true;
}

// =====================
// SAFE SENSOR COMMAND
// =====================
String sendToSensor(const String& cmd, int delayAfter) {

  Wire.beginTransmission(EZO_O2_ADDRESS);
  Wire.write((const uint8_t*)cmd.c_str(), cmd.length());
  Wire.endTransmission();

  delay(delayAfter);

  Wire.requestFrom(EZO_O2_ADDRESS, 32);

  char response[32];
  int i = 0;

  while (Wire.available() && i < 31) {
    char c = Wire.read();

    // keep only printable characters
    if (c >= 32 && c <= 126) {
      response[i++] = c;
    }
  }

  response[i] = '\0';

  String result = String(response);
  result.trim();

  return result;
}

// =====================
// RELAY CONTROL (ACTIVE LOW)
// =====================
void setRelay(bool on) {

  if (on) {

    digitalWrite(RELAY_PIN, LOW);
    state.relayOn = true;

    state.lockoutActive = false;
    state.lockoutStart = 0;

  } else {

    digitalWrite(RELAY_PIN, HIGH);
    state.relayOn = false;

  }
}

// =====================
// START LOCKOUT
// =====================
void startLockout() {

  state.lockoutActive = true;
  state.lockoutStart = millis();

}
// =====================
// LOCKOUT CHECK
// =====================
bool lockoutExpired() {

  if (!state.lockoutActive) {
    return true;
  }

  if (millis() - state.lockoutStart >= lockoutTime) {
    state.lockoutActive = false;
    return true;
  }

  return false;
}

unsigned long lockoutRemaining() {
  if (!state.lockoutActive) return 0;
  if (lockoutExpired()) return 0;
  return lockoutTime - (millis() - state.lockoutStart);
}

// =====================
// CONTROL LOGIC
// =====================
void evaluateRelay() {

  // Technician override
  if (state.manualOverride) {
    setRelay(state.manualRelayState);
    return;
  }

if (state.o2Value < state.setpoint) {

  if (lockoutExpired()) {
    setRelay(true);
  }

} else {

  if (state.relayOn) {
    startLockout();
  }

  setRelay(false);

  }
}

// =====================
// SENSOR READ
// =====================
void readSensor() {
  if (state.sensorSleeping) return;

  String raw = sendToSensor("R", 1200);

  Serial.print("RAW SENSOR RESPONSE: ");
  Serial.println(raw);

  // extract first numeric value safely
  float val = 0.0;

  for (int i = 0; i < raw.length(); i++) {
    if (isDigit(raw[i]) || raw[i] == '.') {
      val = raw.substring(i).toFloat();
      break;
    }
  }

  state.o2Value = val;

  evaluateRelay();

  Serial.println("O2 parsed: " + String(state.o2Value));
}

// =====================
// WEB HANDLERS
// =====================
void handleRoot() {
  if (!checkAuth()) return;
  server.send_P(200, "text/html", index_html);
}

void handleStatus() {
  if (!checkAuth()) return;

  String json =
    "{";
  json += "\"o2\":" + String(state.o2Value, 2) + ",";
  json += "\"setpoint\":" + String(state.setpoint, 2) + ",";
  json += "\"relay\":" + String(state.relayOn ? "true" : "false") + ",";
  json += "\"lockout\":" +
        String(lockoutRemaining()/1000) +
        ",";
  json += "\"manual\":" +
        String(state.manualOverride ? "true" : "false");
  json += "}";

  server.send(200, "application/json", json);
}

// =====================
// COMMANDS (BT + SERIAL)
// =====================
void processCommand(String cmd, Stream &stream) {
  cmd.trim();
  cmd.toUpperCase();

  String response;

  if (cmd == "R") {
    readSensor();
    response = "Read complete";
  }
  else if (cmd == "SLEEP") {
    sendToSensor("Sleep", 300);
    state.sensorSleeping = true;
    response = "Sleeping";
  }
  else if (cmd == "WAKE") {
    Wire.beginTransmission(EZO_O2_ADDRESS);
    Wire.write(0x00);
    Wire.endTransmission();
    state.sensorSleeping = false;
    response = "Awake";
  }
  else {
    response = "Unknown";
  }

  stream.println(response);
  SerialBT.println(response);
}

// =====================
// INPUT CHECK
// =====================
void checkIncoming(Stream &stream) {
  if (stream.available()) {
    String cmd = stream.readStringUntil('\n');
    processCommand(cmd, stream);
  }
}

// =====================
// NEW WEB HANDLERS
// =====================

void handleCalibrate() {

  if (!checkAuth()) return;

  Serial.println("CALIBRATE BUTTON PRESSED");

  sendToSensor("CAL,20.9", 2500);

  delay(1500);

  String verify = sendToSensor("R", 1200);

  Serial.println("Post-Cal Check:");
  Serial.println(verify);

  server.send(
    200,
    "text/plain",
    "Calibration command sent | Current reading: " + verify
  );
}

void handleRelay() {

  if (!checkAuth()) return;

  String stateCmd = server.arg("state");

  if(stateCmd == "on") {

    state.manualOverride = true;
    state.manualRelayState = true;
    manualStartTime = millis();

    setRelay(true);

    server.send(200,"text/plain","Relay ON");
  }

  else if(stateCmd == "off") {

    state.manualOverride = true;
    state.manualRelayState = false;
    manualStartTime = millis();

    setRelay(false);

    server.send(200,"text/plain","Relay OFF");
  }

  else if(stateCmd == "auto") {

    state.manualOverride = false;

    evaluateRelay();

    server.send(200,"text/plain","AUTO MODE");
  }

  else {

    server.send(400,"text/plain","Bad Command");
  }
}

void handleReboot() {

  if (!checkAuth()) return;

  server.send(
    200,
    "text/plain",
    "Rebooting..."
  );

  delay(1000);

  ESP.restart();
}

void handleLogout() {

  if (!checkAuth()) return;

  state.manualOverride = false;

  evaluateRelay();

  server.requestAuthentication();
}

// =====================
// SETUP
// =====================
void setup() {
  Serial.begin(115200);
  Wire.begin(SDA_PIN, SCL_PIN);

  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, HIGH); // OFF (active-low)

  SerialBT.begin("ESP32_O2");

  WiFi.softAP(ap_ssid, ap_password);

  server.on("/", handleRoot);
  server.on("/status", handleStatus);

  server.on("/calibrate", handleCalibrate);
  server.on("/relay", handleRelay);
  server.on("/reboot", handleReboot);
  server.on("/logout", handleLogout);

  server.begin();

  Serial.println(WiFi.softAPIP());

}

// =====================
// LOOP
// =====================
void loop() {

  checkIncoming(Serial);
  checkIncoming(SerialBT);

  server.handleClient();

  if (state.manualOverride) {

  if (millis() - manualStartTime >= manualTimeout) {

    Serial.println("Manual override timeout");

    state.manualOverride = false;
    manualStartTime = 0;
    evaluateRelay();
  }
}

  if (millis() - lastReadTime > readInterval) {
    readSensor();
    lastReadTime = millis();
  }
}
